i < MAX_NON_TERMINALS; i++) { grammar->numProductions[i] = 0; } } // Add a non-terminal to the grammar void addNonTerminal(Grammar *grammar, const char *symbol) { if (grammar->numNonTerminals < MAX_NON_TERMINALS) { strcpy(grammar->nonTerminals[grammar->numNonTerminals], symbol); } // Check if a symbol is a terminal bool isTerminal(Grammar *grammar, const char *symbol) { return findTerminalIndex(grammar, symbol) != -1; } // Compute FIRST sets for all non-terminals void computeFirstSets(Grammar *grammar, Set firstSets[]) { bool changed = true; // Continue until no more changes are made while (changed) { changed = false; // For each non-terminal for (int i = 0; i < grammar->numNonTerminals; i++) { const char *nonTerminal = grammar->nonTerminals[i]; int ntIndex = findNonTerminalIndex(grammar, nonTerminal); // For each production of the non-terminal for (int j = 0; j < grammar->numProductions[ntIndex]; j++) { char *production = grammar->productions[ntIndex][j]; if (strcmp(production, EPSILON) == 0) { // If production is epsilon, add epsilon to FIRST set if (!isInSet(&firstSets[ntIndex], EPSILON)) { addToSet(&firstSets[ntIndex], EPSILON); changed = true; } } else { // Split production into symbols char symbols[MAX_PRODUCTION_LENGTH][MAX_SYMBOL_LENGTH]; int numSymbols = 0; tokenizeProduction(production, symbols, &numSymbols); int symbolIndex = 0; bool allDeriveEpsilon = true; // Process each symbol in the production while (symbolIndex < numSymbols && allDeriveEpsilon) { char *currentSymbol = symbols[symbolIndex]; } allDeriveEpsilon = false; } else if (isNonTerminal(grammar, currentSymbol)) { // If it's a non-terminal, add its FIRST set (except epsilon) int symbolNtIndex = findNonTerminalIndex(grammar, currentSymbol); for (int k = 0; k < firstSets[symbolNtIndex].count; k++) { char *first = firstSets[symbolNtIndex].symbols[k]; if (strcmp(first, EPSILON) != 0 && !isInSet(&firstSets[ntIndex], first)) { addToSet(&firstSets[ntIndex], first); changed = true; } } // Check if this non-terminal can derive epsilon if (!isInSet(&firstSets[symbolNtIndex], EPSILON)) { allDeriveEpsilon = false; } } symbolIndex++; } // If all symbols in the production can derive epsilon, add epsilon to FIRST set if (allDeriveEpsilon && !isInSet(&firstSets[ntIndex], EPSILON)) { addToSet(&firstSets[ntIndex], EPSILON); changed = true; } } } } } } // Compute FOLLOW sets for all non-terminals void computeFollowSets(Grammar *grammar, Set firstSets[], Set followSets[]) { // Add $ to the FOLLOW set of the start symbol int startSymbolIndex = findNonTerminalIndex(grammar, grammar->startSymbol); addToSet(&followSets[startSymbolIndex], END_MARKER); bool changed = true; // Continue until no more changes are made while (changed) { changed = false; // For each production in the grammar for (int i = 0; i < grammar->numNonTerminals; i++) { const char *nonTerminal = grammar->nonTerminals[i]; int ntIndex = findNonTerminalIndex(grammar, nonTerminal); for (int j = 0; j < grammar->numProductions[ntIndex]; j++) { char *production = grammar->productions[ntIndex][j]; if (strcmp(production, EPSILON) == 0) { continue; } // Split production into symbols char symbols[MAX_PRODUCTION_LENGTH][MAX_SYMBOL_LENGTH]; int numSymbols = 0; tokenizeProduction(production, symbols, &numSymbols); // For each symbol in the production for (int k = 0; k < numSymbols; k++) { char *currentSymbol = symbols[k]; // If it's a non-terminal if (isNonTerminal(grammar, currentSymbol)) { int currentSymbolIndex = findNonTerminalIndex(grammar, currentSymbol); // If it's the last symbol if (k == numSymbols - 1) { // Add everything in FOLLOW(nonTerminal) to FOLLOW(currentSymbol) for (int l = 0; l < followSets[ntIndex].count; l++) { if (!isInSet(&followSets[currentSymbolIndex], followSets[ntIndex].symbols[l])) { addToSet(&followSets[currentSymbolIndex], followSets[ntIndex].symbols[l]); changed = true; } } } else { // Compute FIRST of the rest of the string bool allCanDeriveEpsilon = true; int m = k + 1; while (m < numSymbols && allCanDeriveEpsilon) { char *nextSymbol = symbols[m]; if (isTerminal(grammar, nextSymbol)) { // If it's a terminal, add it to FOLLOW and break if (!isInSet(&followSets[currentSymbolIndex], nextSymbol)) { addToSet(&followSets[currentSymbolIndex], nextSymbol); changed = true; } allCanDeriveEpsilon = false; } else if (isNonTerminal(grammar, nextSymbol)) { // Add everything in FIRST(nextSymbol) except epsilon to FOLLOW(currentSymbol) int nextSymbolIndex = findNonTerminalIndex(grammar, nextSymbol); for (int l = 0; l < firstSets[nextSymbolIndex].count; l++) { char *first = firstSets[nextSymbolIndex].symbols[l]; if (strcmp(first, EPSILON) != 0 && !isInSet(&followSets[currentSymbolIndex], first)) { addToSet(&followSets[currentSymbolIndex], first); changed = true; } } // If epsilon is not in FIRST(nextSymbol), we're done if (!isInSet(&firstSets[nextSymbolIndex], EPSILON)) { allCanDeriveEpsilon = false; } } m++; } // If all symbols after k can derive epsilon, add FOLLOW(nonTerminal) to FOLLOW(currentSymbol) if (allCanDeriveEpsilon) { for (int l = 0; l < followSets[ntIndex].count; l++) { if (!isInSet(&followSets[currentSymbolIndex], followSets[ntIndex].symbols[l])) { addToSet(&followSets[currentSymbolIndex], followSets[ntIndex].symbols[l]); changed = true; } } } } } } } } } } // Add a symbol to a set if it's not already there void addToSet(Set *set, const char *symbol) { if (!isInSet(set, symbol) && set->count < MAX_TERMINALS + 1) { strcpy(set->symbols[set->count], symbol); set->count++; } } // Check if a symbol is in a set bool isInSet(Set *set, const char *symbol) { for (int i = 0; i < set->count; i++) { if (strcmp(set->symbols[i], symbol) == 0) { return true; } } return false; } // Print a set void printSet(Set *set) { for (int i = 0; i < set->count; i++) { printf("%s", set->symbols[i]); if (i < set->count - 1) { printf(", "); } } } // Helper function to tokenize a production string into symbols void tokenizeProduction(char *production, char symbols[][MAX_SYMBOL_LENGTH], int *numSymbols) { char *token = strtok(production, " "); *numSymbols = 0; while (token != NULL) { strcpy(symbols[*numSymbols], token); (*numSymbols)++; token = strtok(NULL, " "); } } // Get FIRST set of a production (sequence of symbols) void getFirstOfProduction(Grammar *grammar, Set firstSets[], char *production, Set *resultSet) { resultSet->count = 0; // If production is epsilon, return epsilon if (strcmp(production, EPSILON) == 0) { addToSet(resultSet, EPSILON); return; } // Split production into symbols char symbols[MAX_PRODUCTION_LENGTH][MAX_SYMBOL_LENGTH]; int numSymbols = 0; // Create a copy of production for tokenizing char productionCopy[MAX_PRODUCTION_LENGTH]; strcpy(productionCopy, production); tokenizeProduction(productionCopy, symbols, &numSymbols); bool allCanDeriveEpsilon = true; int i = 0; while (i < numSymbols && allCanDeriveEpsilon) { char *symbol = symbols[i]; if (isTerminal(grammar, symbol)) { addToSet(resultSet, symbol); allCanDeriveEpsilon = false; } else if (isNonTerminal(grammar, symbol)) { int ntIndex = findNonTerminalIndex(grammar, symbol); // Add all elements of FIRST(symbol) except epsilon to result set for (int j = 0; j < firstSets[ntIndex].count; j++) { if (strcmp(firstSets[ntIndex].symbols[j], EPSILON) != 0) { addToSet(resultSet, firstSets[ntIndex].symbols[j]); } } // If epsilon is not in FIRST(symbol), no need to continue if (!isInSet(&firstSets[ntIndex], EPSILON)) { allCanDeriveEpsilon = false; } } i++; } // If all symbols can derive epsilon, add epsilon to result set if (allCanDeriveEpsilon) { addToSet(resultSet, EPSILON); } } // Generate the LL(1) parsing table void generateLL1Table(Grammar *grammar, Set firstSets[], Set followSets[]) { // Add end marker to terminals addTerminal(grammar, END_MARKER);
